PUT /users/:id

Description

Updating a user

Headers

Accept

application/vnd.hairfolio.v1

Content-Type

application/json

Parameters

user[email] required, type: string

The user email

user[first_name] required, type: string

The user first_name

user[last_name] required, type: string

The user last_name

user[account_type] required, type: string

The user account_type ( 'consumer', 'owner', 'stylist', 'brand')

user[password] required, type: string

The user password

user[avatar_url] optional, type: string

The user avatar

user[avatar_cloudinary_id] optional, type: string

The user avatar cloudinary id

user[prof_desc] optional, type: string

The user prof description

user[years_exp] optional, type: string

The user years exp

user[career_opportunity] optional, type: string

The user career opportunities

user[brand_attributes][id] required, type: string

The id of the brand

user[brand_attributes][_destroy] optional, type: string

The set the destroy flag to delete the brand

user[brand_attributes][name] optional, type: string

The name of the brand

user[brand_attributes][address] optional, type: string

The address of the brand

user[brand_attributes][city] optional, type: string

The city of the brand

user[brand_attributes][state] optional, type: string

The state of the brand

user[brand_attributes][website] optional, type: string

The website of the brand

user[brand_attributes][phone] optional, type: string

The phone of the brand

user[salon_attributes][id] required, type: string

The id of the salon to update

user[salon_attributes][_destroy] optional, type: string

Set the destroy tag to delete the salon

user[salon_attributes][name] required, type: string

The name of the salon

user[salon_attributes][address] optional, type: string

The address of the salon

user[salon_attributes][city] optional, type: string

The city of the salon

user[salon_attributes][state] optional, type: string

The state of the salon

user[salon_attributes][zip] optional, type: string

The zip of the salon

user[salon_attributes][phone] optional, type: string

The phone of the salon

user[salon_attributes][website] optional, type: string

The website of the salon

user[experience_ids] optional, type: array

The ids of the expereinces

user[certificate_ids] optional, type: array

The ids of the certificates

user[educations_attributes][id] optional, type: integer

The ids of the education to update

user[educations_attributes][_destroy] optional, type: integer

The set the destroy flag to delete the education

user[educations_attributes][name] optional, type: integer

The name of the education

user[educations_attributes][year_from] optional, type: integer

The year from of the education

user[educations_attributes][degree_id] optional, type: integer

The degree id education

user[educations_attributes][website] optional, type: integer

The webite of the education

user[offerings_attributes][id] optional, type: integer

The ids of the offering to update

user[offerings_attributes][_destroy] optional, type: integer

The set the destroy flag to delete the offering

user[offerings_attributes][price] optional, type: integer

The name of the offering

user[offerings_attributes][service_id] optional, type: integer

The year from of the offering

user[offerings_attributes][category_id] optional, type: integer

The degree id offering
